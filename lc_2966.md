#DIvide arrays into arrays with max difference
```python
class Solution(object):
    def divideArray(self, nums, k):
        """
        :type nums: List[int]
        :type k: int
        :rtype: List[List[int]]
        """
        result=[]
        nums=sorted(nums)
        i=0
        while i < (len(nums)-2) :

            if nums[i+2] - nums[i+1] <= k and nums[i+1] - nums[i] <=k :
                result.append([nums[i],nums[i+1],nums[i+2]])
                i+=3
            else :
                return []

        return result
```
