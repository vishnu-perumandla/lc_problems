# maximum number of words you can type
```c++
class Solution {
public:
    int canBeTypedWords(string text, string brokenLetters) {
        vector<string>s;
        string ss="";
        for(auto i:text){
            if(i==' '){
                s.push_back(ss);
                
                ss="";
            }
            else{
                ss+=i;
            }
            
        }
        s.push_back(ss);
        int count=0;
        vector<int>alp(26,0);
        for(auto i:brokenLetters){
            alp[i-97]+=1;
        }
        for(auto i:s){
            for(auto j:i){
                if(alp[j-97] > 0){
                   count++;
                    break;
                }
            }
        }
        return s.size()-count;
    }
};
```
