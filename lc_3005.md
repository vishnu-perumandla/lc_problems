# Count elements with maximum frequency
```c++
class Solution {
public:
    int maxFrequencyElements(vector<int>& nums) {
        vector<int>v(101,0);
        
        int maxf=0;
        for(int i=0;i<nums.size();i++){
            v[nums[i]]++;
            maxf=max(maxf,v[nums[i]]);
        }
        int count=0;
        for(auto i:v){
            if(i==maxf) count++;
        }
        return count*maxf;
    }
};
```
