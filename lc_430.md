# Flatten a multilevel double linkedlist
```c++
/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* prev;
    Node* next;
    Node* child;
};
*/

class Solution {
public:
    Node* flatten(Node* head) {
        Node* current=head;
        Node* temp=head;
        Node* child_H;

        while(current){
            temp=current;
            if(temp->child){
                child_H=temp->child;
                temp=temp->child;
                while(temp->next) temp=temp->next;
                temp->next=current->next;
                if(current->next)current->next->prev=temp;
                current->next=child_H;
                child_H->prev=current;
                current->child=NULL;
            }
            current=current->next;
        }
        return head;
    }
};
```
