# sort matrix by diagonals
```python
class Solution(object):
    def sortMatrix(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: List[List[int]]
        """
        m=len(grid)
        n=len(grid[0])
        
        arr=[]
        if m > 1 :
            
            for j in range(1,n) :
                i=0
                sj=j
                while i < m and j < n :
                    arr.append(grid[i][j])
                    i+=1
                    j+=1
                arr=sorted(arr)
                k=0
                i=0
                j=sj
                while i < m and j < n :
                    grid[i][j]=arr[k]
                    i+=1
                    k+=1
                    j+=1
                arr=[]

            for i in range(0,m) :
                j=0
                si=i
                while i < m and j < n :
                    arr.append(grid[i][j])
                    i+=1
                    j+=1
                arr=sorted(arr,reverse=True)
                k=0
                j=0
                i=si
                while i < m and j < n :
                    grid[i][j]=arr[k]
                    i+=1
                    k+=1
                    j+=1
                arr=[]

        return grid
```
