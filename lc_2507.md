# Smallest value ofter replacing with sum of primefactors
```python
class Solution(object):
    def smallestValue(self, n):
        """
        :type n: int
        :rtype: int
        """
        def is_prime(n) :
            
            for i in range(2,(n//2)+1) :
                if n % i == 0:
                    return False

            return True

        
        def recur(n) :

            if is_prime(n) :
                return n
            r=0
            i=2
            arr=[]
            n1=n
            while i < (n//2)+1 :
                if n%i == 0 :
                    if is_prime(i) :
                        r+=i
                        arr.append(i)
                        n1/=i
                i+=1
            
            while n1 > 1 :
                for i in arr :
                    if n1%i==0 :
                        n1/=i
                        r+=i
                
            if r==n :
                return n
            return recur(r)

        return recur(n)
```
