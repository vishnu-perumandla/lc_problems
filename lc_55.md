# Jumpgame
```c++
class Solution {
public:
    bool canJump(vector<int>& nums) {
        int position=0;
        int ml=nums[0];
        if(nums.size()-1==0) return true;
        while (ml!=position) {
            if(ml >= nums.size()-1) return true;
            int max[]={0,0} ;//for storing maxjum and index of that maxjum
            for(int i=position+1;i <= ml;i++){
                if (nums[i] >= max[0]){
                    max[0]=nums[i];
                    max[1]=i;
                } 

            }
            int start=max[1]+1;
            for(int i=start;i<=ml;i++){
                if(max[0]+max[1] > ml) break;
                max[0]=nums[i];
                max[1]=i;
            }
            position=max[1];
            ml=position+max[0];
            
        }
        return false;
    }
};
```
