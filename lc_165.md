# Compare version numbers
```c++
class Solution {
public:
    int compareVersion(string version1, string version2) {
        vector<int>v1;
        vector<int>v2;
        
        int i=0;
        while(i < version1.size()){
            string String="";
            while(i<version1.size() && version1[i]!='.'){
                String+=version1[i];
                i++;
            }
            v1.push_back(stoi(String));
            i++;
        }
        i=0;
        while(i < version2.size()){
            string string="";
            while(i<version2.size() && version2[i]!='.'){
                string+=version2[i];
                i++;
            }
            v2.push_back(stoi(string));
            i++;
        }
        
        int v1s=v1.size();
        int v2s=v2.size();
        if(v1s < v2s){
            for(int i=0;i<v2s-v1s;i++){
                v1.push_back(0);
            }
        }
        if(v1s > v2s){
            for(int i=0;i<v1s-v2s;i++){
                v2.push_back(0);
            }
        }
        
        for(int i=0;i<v1.size();i++){
            
            if(v1[i] > v2[i]) return 1;
            else if(v1[i] < v2[i]) return -1;
            

        }
        return 0;
    }
};
```
