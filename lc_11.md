# Containr with most water
```c++
class Solution {
public:
    int maxArea(vector<int>& height) {
        
        int left=0;
        int right=height.size()-1;
        long long int area=0;

        while(left < right){
            long long int l=min(height[left],height[right]);
            long long int b=right - left ;
            if (l*b > area){
                area=l*b;
            }

            if (height[left] < height[right] ){
                left+=1;
            }
            else{
                right-=1;
            }
        }
        return area;
    }
};
```
