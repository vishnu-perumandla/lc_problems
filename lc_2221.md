# Find triangular sum of an array
```c++
class Solution {
public:
    int triangularSum(vector<int>& nums) {
        vector<int>t1=nums;
        vector<int>t2;

        while(t1.size() > 1 || t2.size() > 1){
            if( t1.size() > 1){
                //t2.clear();
            for(int i=0;i<t1.size()-1;i++){
                t2.push_back((t1[i]+t1[i+1])%10);
            }
            }
            if(t1.size() > 1) t1.clear();
            
            if(t2.size() > 1){
            for(int i=0;i<t2.size()-1;i++){
                t1.push_back((t2[i]+t2[i+1]) % 10);
            }
            }
            if(t2.size() > 1) t2.clear();
        }
        if(t1.size() == 1) return t1[0];
        if(t2.size() == 1) return t2[0];
        return 0;
    }
};
```
