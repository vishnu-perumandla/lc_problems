# Reorder list 
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    void reorderList(ListNode* head) {
        vector<int>v;
        ListNode* temp=head;
        while(temp){
            v.push_back(temp->val);
            temp=temp->next;
        }
        int f=0;
        int l=v.size()-1;
        temp=head;
        int i=1;
        while(temp){
            if(i%2!=0){
                temp->val=v[f++];
                
            }
            else{
                temp->val=v[l--];
                
            }
            i++;
            temp=temp->next;
        }
        
    }
};
```
