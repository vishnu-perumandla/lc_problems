# Reverse inkedlist II
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    void swap(ListNode* a,ListNode* b){
        int t=a->val;
        a->val=b->val;
        b->val=t;
    }
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        vector<ListNode*>lr;
        ListNode* temp=head;
        int i=1;
        while(temp){
            if(i >=left and i<=right){
                lr.push_back(temp);
            }
            temp=temp->next;
            i++;
            
            if(i > right) break;
        }
        int size=lr.size()-1;
        ListNode* l=lr[0];
        ListNode* r=lr[size--];
        while(l<r){
            swap(l,r);
            l=l->next;
            r=lr[size--];
        }
        return head;
    }
};
```
