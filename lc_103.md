# Binary tree level order traversal
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>>v;
        vector<int>sv;
        queue<TreeNode*>q1;
        queue<TreeNode*>q2;
        TreeNode* temp;
        if(!root) return v;
        q1.push(root);
        sv.push_back(root->val);
        v.push_back(sv);
        sv.clear();

        while(!q1.empty() || !q2.empty()){
            while(!q1.empty()){
                temp=q1.front();
                q1.pop();
                if(temp->left){
                    q2.push(temp->left);
                    sv.push_back(temp->left->val);
                }
                if(temp->right){
                    q2.push(temp->right);
                    sv.push_back(temp->right->val);
                }
            }
            if(!sv.empty()){
                v.push_back(sv);
            }
            sv.clear();
            while(!q2.empty()){
                temp=q2.front();
                q2.pop();
                if(temp->left){
                    q1.push(temp->left);
                    sv.push_back(temp->left->val);
                }
                if(temp->right){
                    q1.push(temp->right);
                    sv.push_back(temp->right->val);
                }
            }
            if(!sv.empty()){
                v.push_back(sv);
            }
            sv.clear();
        }
        return v;
    }
};
```
