# Zero array transformation I
``` python
class Solution(object):
    def isZeroArray(self, nums, queries):
        """
        :type nums: List[int]
        :type queries: List[List[int]]
        :rtype: bool
        """
        diff=[0] * (len(nums)+1)
        

        for i in queries :

            diff[i[0]]+=1
            diff[i[1]+1]-=1

        applied=0
        for i in range(len(nums)) :
            
            applied+=diff[i]
            if nums[i] > applied :
                return False

        return True
```
