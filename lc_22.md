# Generate peranthesis
```c++
class Solution {
public:
    struct node {
        node* left;
        node* right;
        string str;
        int openCount;
        int closeCount;
        node(string s, int openC, int closeC) {
            left = NULL;
            right = NULL;
            str = s;
            openCount = openC;
            closeCount = closeC;
        }
    };

    vector<string> generateParenthesis(int n) {
        vector<string> result;
        queue<node*> q;

        node* root = new node("(", 1, 0);
        q.push(root);

        while (!q.empty()) {
            node* temp = q.front();
            q.pop();

            if (temp->openCount == n && temp->closeCount == n) {
                result.push_back(temp->str);
                continue;
            }

            // Add '(' if we can
            if (temp->openCount < n) {
                temp->left = new node(temp->str + "(", temp->openCount + 1, temp->closeCount);
                q.push(temp->left);
            }

            // Add ')' if we can
            if (temp->closeCount < temp->openCount) {
                temp->right = new node(temp->str + ")", temp->openCount, temp->closeCount + 1);
                q.push(temp->right);
            }
        }
        return result;
    }
};
```
