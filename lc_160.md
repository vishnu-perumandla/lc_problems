# finding the intersection node
```c++
class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        vector<ListNode*> n1, n2;
        for (auto p = headA; p; p = p->next) n1.push_back(p);
        for (auto p = headB; p; p = p->next) n2.push_back(p);

        int i = n1.size() - 1, j = n2.size() - 1;
        ListNode* intersection = nullptr;

        while (i >= 0 && j >= 0 && n1[i] == n2[j]) {
            intersection = n1[i];
            i--; j--;
        }

        return intersection;
    }
};
```
