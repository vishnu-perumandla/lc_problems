# Populating next right pointer in each node
```c++
/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* left;
    Node* right;
    Node* next;

    Node() : val(0), left(NULL), right(NULL), next(NULL) {}

    Node(int _val) : val(_val), left(NULL), right(NULL), next(NULL) {}

    Node(int _val, Node* _left, Node* _right, Node* _next)
        : val(_val), left(_left), right(_right), next(_next) {}
};
*/

class Solution {
public:
    Node* connect(Node* root) {
        
        queue<Node*>q;
        
        Node* temp;
        if(!root) return root;
        
        q.push(root);
        while(!q.empty()){
            
            queue<Node*>tq;
            while(!q.empty()){
                temp=q.front();
                q.pop();
                if(!q.empty())temp->next=q.front();
                if(temp->left){
                    tq.push(temp->left);
                }
                if(temp->right){
                    tq.push(temp->right);
                }
            }
            q=tq;
        }
        return root;
    }
};
```
