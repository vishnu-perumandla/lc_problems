#number of zero_filled subarrays
```class Solution(object):
    def zeroFilledSubarray(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        rn_count=[]
        
        i=0

        while i < len(nums) :
            c=0

            while i < len(nums) and nums[i]==0 :
                c+=1
                i+=1
            if c != 0 :
                rn_count.append(c)

            i+=1

        result=0
        n=1
        print(rn_count)
        while True :
            c=0

            for i in rn_count :
                
                if i >= n :
                    c+=i-n+1

            if c==0 :
                break
            result+=c
            n+=1
            

        return result
```
