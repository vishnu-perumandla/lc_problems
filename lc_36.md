class Solution(object):
    def isValidSudoku(self, board):
        """
        :type board: List[List[str]]
        :rtype: bool
        """
        for i in range(9) :
            for j in range(9):

                col=[]
                if board[i][j]=='.' :
                    continue

                if board[i][j] in board[i][j+1:]:
                    return False

                # column check
                col = [board[x][j] for x in range(i+1, 9)]
                if board[i][j] in col:
                    return False
                
                c = 0
                for k in range((i//3)*3, (i//3)*3 + 3):
                    for l in range((j//3)*3, (j//3)*3 + 3):
                        if board[i][j] == board[k][l]:
                            c += 1
                if c > 1:
                    return False


        return True
