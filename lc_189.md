# rotate array 
```c++
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        int size=nums.size();
        k=k%size;
        vector<int>t;
        t.insert(t.begin(),nums.end()-k,nums.end());
        nums.insert(nums.begin(),t.begin(),t.end());
        nums.erase(nums.end()-k,nums.end());
        
    }
};
```
