# Find N uniquiue integers whose sum is up to zero
```c++
class Solution {
public:
    vector<int> sumZero(int n) {
        vector<int>result;
        for(int i=-n/2;i<=n/2;i++){
            if(n%2==0 && i==0) continue;
            else{
                result.push_back(i);
            }
        }
        return result;
    }
};
```
