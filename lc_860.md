# Lemonade change
```python
class Solution(object):
    def lemonadeChange(self, bills):
        """
        :type bills: List[int]
        :rtype: bool
        """
        collection=[]        

        for i in range(len(bills)) :
            collection.append(bills[i])

            if bills[i]==10 :
                if 5 in collection :
                    collection.remove(5)
                else :
                    return False

            elif bills[i]==20 :
                if 5 in collection :
                    if 10 in collection :
                        collection.remove(10)
                        collection.remove(5)
                    elif collection.count(5) >= 3 :
                        collection.remove(5)
                        collection.remove(5)
                        collection.remove(5)
                    else :
                        return False
                else :
                    return False

        return True

```
