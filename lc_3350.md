# Adjacent increasing subarray detection II
```c++
class Solution {
public:
    int maxIncreasingSubarrays(vector<int>& nums) {
        int s=nums.size();
        vector<int>v(s,1);
        cout<<v[0]<<" ";
        for(int i=1;i<s;i++){
            if(nums[i]>nums[i-1]) v[i]=v[i-1]+1;
            cout<<v[i]<<" ";
        }
        int maxv=0;
        for(int i=1;i<s;i++){
            maxv=max(maxv,v[i]/2);
            if(v[i] > v[i-1]) continue;
            
            int left=v[i-1];
            int right=1;
            int j=i+1;
            while(j<s && v[j]>v[j-1]){
                j++;
                right++;
            }
                
                maxv=max(maxv,min(left,right));
            
        }
        return maxv;
    }
};
```
