# Finding resultant array ofter removing anagrams
```c++
class Solution {
public:
    vector<string> removeAnagrams(vector<string>& words) {
        
        int i=1;
        int s=words.size();
        while(i < s){
            vector<int>v(26,0);
            for(auto j:words[i]) v[j-97]++;
            int w=0;
            for(auto j:words[i-1]){
                if(v[j-97]==0 ){
                    cout<<j<<" ";
                    w=1;
                    break;
                }
                else v[j-97]--;
                
            }
            if(w==0 && words[i].size()==words[i-1].size()){
                words.erase(words.begin()+i);
                s--;
            }
            else i++;
        }

        return words;
    }
};
```
