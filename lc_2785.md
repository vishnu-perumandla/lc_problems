# Sort the ovels in string
```c++
class Solution {
public:
    string sortVowels(string s) {
       priority_queue<int>vo;

        for(auto i:s ){
            if(i=='A' || i=='E' || i=='I' || i=='O' || i=='U'|| i=='a' || i=='e' || i=='i' || i=='o' || i=='u'){
                vo.push(i);
            }
        }

        for(int i=s.size()-1 ;i>=0 ;i--){
            if(s[i]=='A' || s[i]=='E' || s[i]=='I' || s[i]=='O' || s[i]=='U'|| s[i]=='a' || s[i]=='e' || s[i]=='i' || s[i]=='o' || s[i]=='u'){
                s[i]=vo.top();
                vo.pop();
            }
        }
        return s;
    }
};
```
