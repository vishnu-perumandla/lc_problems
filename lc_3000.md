# Maximum area of longest diagonal Rectangle

class Solution(object):
    def areaOfMaxDiagonal(self, dimensions):
        """
        :type dimensions: List[List[int]]
        :rtype: int
        """
        diagonals=[]

        for i in dimensions :
            diagonals.append( sqrt(i[0]*i[0] + i[1]*i[1]) )
        
        if diagonals.count(max(diagonals)) > 1 :
            maxarea=0
            print(diagonals)
            for i in dimensions :
                if sqrt(i[0]*i[0] + i[1]*i[1]) == max(diagonals) :
                    maxarea=max(maxarea,i[0]*i[1])

            return maxarea
        
        ind=diagonals.index(max(diagonals))
        return dimensions[ind][0] * dimensions[ind][1]
