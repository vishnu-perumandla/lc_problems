# kth smallest element in the sorted matrix
```python
class Solution(object):
    def kthSmallest(self, matrix, k):
        """
        :type matrix: List[List[int]]
        :type k: int
        :rtype: int
        """
        arr=[]

        for i in range(len(matrix)) :
            for j in range(len(matrix[i])) :
                arr.append(matrix[i][j])

        arr=sorted(arr)

        return arr[k-1]

```
