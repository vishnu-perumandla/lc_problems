# Findout most frequent ovels and consonents
```c++
class Solution {
public:
    int maxFreqSum(string s) {
        vector<int>alp(26,0);
        for(auto i:s){
            alp[i-97]++;
        }
        int cmax=0;
        int omax=0;
        for(int i=0;i<26;i++){
            
                if(i+97=='a' || i+97=='e' || i+97=='i' || i+97=='o' || i+97=='u'){
                    omax=max(omax,alp[i]);
                }
                else{
                    cmax=max(cmax,alp[i]);
                }
            
        }
        return omax+cmax;
    }
};
```
