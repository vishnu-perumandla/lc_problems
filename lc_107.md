# Binary levelorder traversal II
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrderBottom(TreeNode* root) {
        vector<int>sv;
        vector<vector<int>>v;
        
        queue<TreeNode*>q;
        TreeNode* temp;
        if(!root) return v;
        sv.push_back(root->val);
        v.push_back(sv);
        q.push(root);
        while(!q.empty()){
            int size=q.size();
            vector<int>sv;
            for(int i=0;i<size;i++){
                temp=q.front();
                q.pop();
                if(temp->left){
                    sv.push_back(temp->left->val);
                    q.push(temp->left);
                }
                if(temp->right){
                    sv.push_back(temp->right->val);
                    q.push(temp->right);
                }
            }
            if(!sv.empty())v.push_back(sv);
        }
        vector<vector<int>>r;
        for(int i=v.size()-1;i>=0;i--){
            r.push_back(v[i]);
        }
        return r;
    }
};
```
