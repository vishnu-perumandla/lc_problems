# Successfull pairs of spells and potions
```c++
class Solution {
public:
    int b_s(const vector<int>& potions, long long success, long long spell) {
        int left = 0, right = potions.size();
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (1LL * spell * potions[mid] >= success)
                right = mid;
            else
                left = mid + 1;
        }
        return left;
    }

    vector<int> successfulPairs(vector<int>& spells, vector<int>& potions, long long success) {
        sort(potions.begin(), potions.end());
        int m = potions.size();
        vector<int> ans;
        ans.reserve(spells.size());

        for (long long spell : spells) {
            if (spell * potions.back() < success) {
                ans.push_back(0);
                continue;
            }
            int idx = b_s(potions, success, spell);
            ans.push_back(m - idx);
        }

        return ans;
    }
};
```
