# Adjacent increasing subarray II
```c++
class Solution {
public:
    bool hasIncreasingSubarrays(vector<int>& nums, int k) {
        int s=nums.size();
        if(k==1) return true;
        vector<int>v(s,1);
        cout<<v[0]<<" ";
        for(int i=1;i<s;i++){
            if(nums[i]>nums[i-1]) v[i]=v[i-1]+1;
            cout<<v[i]<<" ";
        }
        for(int i=1;i<s;i++){
            if(v[i]/2 >= k)return true;
            if(v[i]<v[i-1] && v[i-1] >= k && i-1+k<s && v[i-1+k]>=k) return true;
        }
        return false;
    }
};
```
