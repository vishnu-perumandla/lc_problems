# Lowest commen ansister of a BST
```c++
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */

class Solution {
public:
    TreeNode* r;
    void BST(TreeNode* root,int a,int b){
        if(root){
            if(root->val > a && root->val >b) BST(root->left,a,b);
            else if(root->val < a && root->val <b) BST(root->right,a,b);
            else {
                 r=root;
                 return ;
            }
        }
    }
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
         BST(root,p->val,q->val);
         return r;
    }
};
```
