        if(root) {
        if(root->left) in_order(root->left);
        v.push_back(root->val);
        if(v.size() >=2){
            if(v[v.size()-1] <= v[v.size()-2]){
                k=0;
                return;
            }
        }
        if(root->right) in_order(root->right);
        }
        
    }
