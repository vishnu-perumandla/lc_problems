# convert integer to sum of two No-zero integers
```c++
class Solution {
public:
    bool checking(int a,int b){
        
        while(a>0){
            if(a%10==0) return true;
            a/=10;
        }
        while(b > 0){
            if(b%10==0) return true;
            b/=10;
        }
        return false;
    }
    vector<int> getNoZeroIntegers(int n) {
        vector<int>v;
        int a=n-1;
        int b=1;
        while(checking(a,b)){
            a-=1;
            b+=1;
        }
        v.push_back(a);
        v.push_back(b);
        return v;
    }
};
```
