# Search in rotated sorted array
```c++
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int ind=0;
        int n=nums.size();
        for(int i=1;i<nums.size();i++){
            if(nums[i] < nums[i-1]){
                ind=i;
                break;
            }
            
        }
        cout<<ind;
        if(ind > 0){
            int left=ind;
            int right=n-1;
            while(left <= right){
                int mid=left+(right-left)/2;
                if(nums[mid] < target) left=mid+1;
                else if(nums[mid] > target) right=mid-1;
                else return mid;
            }
            left=0;
            right=ind-1;
            while(left <= right){
                int mid=left+(right-left)/2;
                if(nums[mid] < target) left=mid+1;
                else if(nums[mid] > target) right=mid-1;
                else return mid;
            }
        }
        int left=0;
        int right=n-1;
        while(left <= right){
            int mid=left+(right-left)/2;
            if(nums[mid] < target) left=mid+1;
            else if(nums[mid] > target) right=mid-1;
            else return mid;
        }
        return -1;
    }
};
```
