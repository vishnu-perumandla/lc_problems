class Solution(object):
    def maximumGroups(self, grades):
        """
        :type grades: List[int]
        :rtype: int
        """
        groups=0
        n=len(grades)
        temp=1
        
        grades=sorted(grades)
        while n > 0 :
            i=0
            if len(grades) < temp :
                break
            for i in range(temp) :
                print(grades.pop(0))
                
                n-=1
            groups+=1
            temp+=1
        return groups
