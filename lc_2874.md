# Max value of a ordered triplet II

```python
class Solution(object):
    def maximumTripletValue(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        result=0
        le=len(nums)
        

        suffmax=[0]*le
        suffmax[-1]=nums[-1]

        for i in range(le-2,-1,-1) :
            suffmax[i]=max(nums[i],suffmax[i+1])

        i=nums[0]
        for m in range(1,len(nums)-1) :
            if nums[m] < i :
                k=suffmax[m+1]
                result=max(result,(i-nums[m])*k)
            
            i=max(nums[m],i)

        return result
```
