# Largest perimeter triangle
```c++
class Solution {
public:
    int largestPerimeter(vector<int>& nums) {
        sort(nums.begin(),nums.end());
        int n=nums.size();
        
        int k=n-1;
        int perimeter=0;
        for(k;k > 1;k--){
            int j=k-1;
            if(nums[j-1]+nums[j] > nums[k]){
                perimeter=max(perimeter,nums[j-1]+nums[j]+nums[k]);
            }
        }
        return perimeter;
    }
};
```
